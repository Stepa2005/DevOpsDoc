1. docker build -t my-app:1.0  . - делает сборку образа если Dockerfile находится в текущей директории
2. docker run --name my-container -d my-app:1.0 - запуск контейнера (-d означает в фоновом режиме) и присваивание ему имени my-container
3. docker ps - проверка запущенных контейнеров
4. docker logs my-container - просмотр лога контейнера с именем my-container
5. docker exec -it my-container /bin/bash - выполнение команд внутри контейнера
6. docker stop my-container - остановка работы контейнера с именем my-container
7. docker rm my-container - удаление контейнера с именем my-container
